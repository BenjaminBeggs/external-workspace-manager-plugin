The object returned by the exwsAllocate step.

Basic usage: 

    def extWorkspace = exwsAllocate diskPoolId: 'diskpool1'
    node ('linux') {
        exws (extWorkspace) {
            scm checkout
            sh 'mvn clean install -DskipTests'
        }
    }
    node ('test') {
        exws (extWorkspace) {
            sh 'mvn test'
        }
    }
